```js
// suit: 'm'(万), 'p'(筒), 's'(索), 'z'(字)
// rank: 1-9 for number suits, 1-7 for z (東南西北白發中)
// aka: true for 赤牌

export const TILE_TYPES = [];

// 万子 m1-m9
for (let r = 1; r <= 9; r++) {
  TILE_TYPES.push({ suit: 'm', rank: r });
}
// 筒子 p1-p9
for (let r = 1; r <= 9; r++) {
  TILE_TYPES.push({ suit: 'p', rank: r });
}
// 索子 s1-s9
for (let r = 1; r <= 9; r++) {
  TILE_TYPES.push({ suit: 's', rank: r });
}
// 字牌 z1-z7 (東南西北白發中)
for (let r = 1; r <= 7; r++) {
  TILE_TYPES.push({ suit: 'z', rank: r });
}

export function tileKey(suit, rank, aka = false) {
  return `${suit}${rank}${aka ? 'a' : ''}`;
}

export function isAkahai(tile) {
  return tile.aka === true;
}

export function isYaochuu(tile) {
  // 幺九牌: 1和9的数牌 + 所有字牌
  if (tile.suit === 'z') return true;
  return tile.rank === 1 || tile.rank === 9;
}

export function createTileSet() {
  const tiles = [];
  let id = 0;

  for (const type of TILE_TYPES) {
    if (type.suit === 'm' && type.rank === 5) {
      // 五万: 3张普通 + 1张赤
      for (let i = 0; i < 3; i++) {
        tiles.push({ id: id++, suit: type.suit, rank: type.rank, aka: false, key: tileKey(type.suit, type.rank, false) });
      }
      tiles.push({ id: id++, suit: type.suit, rank: type.rank, aka: true, key: tileKey(type.suit, type.rank, true) });
    } else if (type.suit === 'p' && type.rank === 5) {
      // 五筒: 3张普通 + 1张赤
      for (let i = 0; i < 3; i++) {
        tiles.push({ id: id++, suit: type.suit, rank: type.rank, aka: false, key: tileKey(type.suit, type.rank, false) });
      }
      tiles.push({ id: id++, suit: type.suit, rank: type.rank, aka: true, key: tileKey(type.suit, type.rank, true) });
    } else if (type.suit === 's' && type.rank === 5) {
      // 五索: 3张普通 + 1张赤
      for (let i = 0; i < 3; i++) {
        tiles.push({ id: id++, suit: type.suit, rank: type.rank, aka: false, key: tileKey(type.suit, type.rank, false) });
      }
      tiles.push({ id: id++, suit: type.suit, rank: type.rank, aka: true, key: tileKey(type.suit, type.rank, true) });
    } else {
      // 其他: 4张普通
      for (let i = 0; i < 4; i++) {
        tiles.push({ id: id++, suit: type.suit, rank: type.rank, aka: false, key: tileKey(type.suit, type.rank, false) });
      }
    }
  }

  return tiles;
}

// 字牌名称映射
export const Z_TILE_NAMES = {
  1: '東', 2: '南', 3: '西', 4: '北',
  5: '白', 6: '發', 7: '中',
};

// 数牌汉字映射
export const M_RANK_NAMES = { 1:'一', 2:'二', 3:'三', 4:'四', 5:'五', 6:'六', 7:'七', 8:'八', 9:'九' };

export function tileDisplayName(tile) {
  if (tile.suit === 'z') return Z_TILE_NAMES[tile.rank];
  if (tile.suit === 'm') return `${M_RANK_NAMES[tile.rank]}萬`;
  if (tile.suit === 'p') return `${tile.rank}筒`;
  if (tile.suit === 's') return `${tile.rank}索`;
```

```js
// xorshift128+ pseudorandom number generator
// Seeded shuffle for reproducible game sequences

export function createRng(seed) {
  let s0 = seed;
  let s1 = seed ^ 0x6c62272e7d2b5a3dn;

  // Warm up the generator
  for (let i = 0; i < 64; i++) {
    const result = s0 + s1;
    s1 ^= s0;
    s0 = (s1 ^ ((s1 << 23) | (s1 >>> 41)) ^ (s1 >>> 17)) | 0;
    s1 = (s1 ^ ((s1 >>> 18))) | 0;
    s0 = Math.imul(s0, 0x6c62272e7d2b5a3d);
    s1 = Math.imul(s1, 0x5b8a8a98d3d4d4d3);
  }

  return function next() {
    const result = (s0 + s1) & 0x7fffffff;
    s1 ^= s0;
    s0 = ((s1 ^ ((s1 << 23) | (s1 >>> 41)) ^ (s1 >>> 17)) & 0x7fffffff);
    s1 = ((s1 ^ (s1 >>> 18)) & 0x7fffffff);
    return result;
  };
}

// Fisher-Yates shuffle using seeded RNG
export function xorshiftShuffle(array, seed) {
  const rng = createRng(seed);
  const result = [...array];
  for (let i = result.length - 1; i > 0; i--) {
    const j = rng() % (i + 1);
    [result[i], result[j]] = [result[j], result[i]];
  }
  return result;
}
```

```js
export const WINDS = ['東', '南', '西', '北'];

export const INITIAL_POINTS = 25000;

export const RIICHI_COST = 1000;

export const GAME_LENGTH = [
  { wind: '東', number: 1 }, { wind: '東', number: 2 },
  { wind: '東', number: 3 }, { wind: '東', number: 4 },
  { wind: '南', number: 1 }, { wind: '南', number: 2 },

export function isMenzenchin(calls) {
  return calls.every(c => c.type === 'ankou' || c.type === 'kakan');

// 立直条件: 门前清 + 听牌 + 点数>=1000
export function canRiichi(calls, isTenpai, points) {
  return isMenzenchin(calls) && isTenpai && points >= RIICHI_COST;

// 宝牌翻牌: 指示牌的下一张就是宝牌
export function doraFromIndicator(indicator) {
  if (indicator.suit === 'z') {
    // 字牌循环: 東→南→西→北→東, 白→發→中→白
    if (indicator.rank <= 4) return { suit: 'z', rank: indicator.rank % 4 + 1 };
    return { suit: 'z', rank: indicator.rank % 7 + 5 <= 7 ? indicator.rank % 7 + 5 : 5 };
  }
  // 数牌: rank循环 1→2→...→9→1
  return { suit: indicator.suit, rank: indicator.rank % 9 + 1 };
}

// 吃碰杠操作优先级
export const ACTION_PRIORITY = {
  'tsumo': 5,    // 自摸和
  'ron': 5,      // 荣和
  'kan': 4,      // 杠(明杠/加杠)
  'pon': 3,      // 碰
```

```js
// 手牌管理模块

import { createTileSet } from './tiles.js';

const SUIT_ORDER = { 'm': 0, 'p': 1, 's': 2, 'z': 3 };

export function createHand(tiles) {
  return {
    tiles: sortHand(tiles),
    drawnTile: null,
    calls: [],     // 吃碰杠的副露
    riichi: false,
    riichiTurn: null, // 立直时打出的牌的巡目
  };
}

export function sortHand(tiles) {
  return [...tiles].sort((a, b) => {
    const suitDiff = SUIT_ORDER[a.suit] - SUIT_ORDER[b.suit];
    if (suitDiff !== 0) return suitDiff;
    return a.rank - b.rank;
  });
}

export function removeTileFromHand(hand, tileId) {
  // Can remove from main hand or drawn tile
  if (hand.drawnTile && hand.drawnTile.id === tileId) {
    return { ...hand, drawnTile: null };
  }
  return {
    ...hand,
    tiles: hand.tiles.filter(t => t.id !== tileId),
  };
}

export function addTileToHand(hand, tile) {
  return { ...hand, drawnTile: tile };
}

// 获取手牌中所有牌（包括自摸牌），排序后的
export function getAllHandTiles(hand) {
  const all = hand.drawnTile ? [...hand.tiles, hand.drawnTile] : [...hand.tiles];
  return sortHand(all);
}

// 判断是否可以吃（只有下家可以吃）
export function canChi(hand, discardedTile, nextPlayerWind) {
  if (discardedTile.suit === 'z') return []; // 字牌不能吃
  if (hand.calls.length > 0 && !hand.calls.every(c => c.type === 'ankou' || c.type === 'kakan')) {
    // 非门前清也可以吃，只是不能立直
  }
  // 吃需要手中有与出牌同花色的相邻两张牌
  const combos = [];
  const sameSuit = hand.tiles.filter(t => t.suit === discardedTile.suit && !t.aka);
  const ranks = sameSuit.map(t => t.rank);

  // 左吃: 出牌+出牌+1+出牌+2 (出牌在最低位)
  if (ranks.includes(discardedTile.rank + 1) && ranks.includes(discardedTile.rank + 2)) {
    combos.push({ type: 'chi_left', tiles: [discardedTile.rank, discardedTile.rank + 1, discardedTile.rank + 2] });
  }
  // 中吃: 出牌-1+出牌+出牌+1 (出牌在中间位)
  if (ranks.includes(discardedTile.rank - 1) && ranks.includes(discardedTile.rank + 1)) {
    combos.push({ type: 'chi_mid', tiles: [discardedTile.rank - 1, discardedTile.rank, discardedTile.rank + 1] });
  }
  // 右吃: 出牌-2+出牌-1+出牌 (出牌在最高位)
  if (ranks.includes(discardedTile.rank - 2) && ranks.includes(discardedTile.rank - 1)) {
    combos.push({ type: 'chi_right', tiles: [discardedTile.rank - 2, discardedTile.rank - 1, discardedTile.rank] });
  }

  return combos;
}

// 判断是否可以碰
export function canPon(hand, discardedTile) {
  const count = hand.tiles.filter(t => t.suit === discardedTile.suit && t.rank === discardedTile.rank).length;
  return count >= 2;
}

// 判断是否可以明杠
export function canMinkan(hand, discardedTile) {
  const count = hand.tiles.filter(t => t.suit === discardedTile.suit && t.rank === discardedTile.rank).length;
  return count >= 3;
}

// 判断是否可以加杠（已有碰的基础上第4张）
export function canKakan(hand) {
  return hand.calls.filter(c => c.type === 'minpon' && hand.drawnTile &&
    hand.drawnTile.suit === c.tiles[0].suit && hand.drawnTile.rank === c.tiles[0].rank);
}
```

```js
// 判断一组14张牌是否构成和牌
// modes: checkStandard (4面子1雀头), checkChitoi (七对子), checkKokushi (国士无双)
export function isWinningHand(tiles, checkChitoi = true, checkKokushi = true) {
  if (tiles.length !== 14) return false;

  // Check kokushi first (13 wait pattern)
  if (checkKokushi && isKokushi(tiles)) return true;

  // Check seven pairs
  if (checkChitoi && isChitoi(tiles)) return true;

  // Check standard 4 mentsu + 1 jantai
  return findMentsuDecompositions(tiles).length > 0;
}

// 七对子判定
function isChitoi(tiles) {
  const counts = {};
  tiles.forEach(t => {
    const key = `${t.suit}${t.rank}`;
    counts[key] = (counts[key] || 0) + 1;
  });
  const pairs = Object.entries(counts).filter(([k, v]) => v === 2);
  return pairs.length === 7;
}

// 国士无双判定
function isKokushi(tiles) {
  const yaochuuKeys = [
    'm1', 'm9', 'p1', 'p9', 's1', 's9',
    'z1', 'z2', 'z3', 'z4', 'z5', 'z6', 'z7'
  ];
  const counts = {};
  tiles.forEach(t => {
    const key = `${t.suit}${t.rank}`;
    counts[key] = (counts[key] || 0) + 1;
  });

  // Each yaochuu type must appear at least once, and exactly one type appears twice
  let doubleCount = 0;
  for (const key of yaochuuKeys) {
    if (!counts[key]) return false;
    if (counts[key] === 2) doubleCount++;
    if (counts[key] > 2) return false;
  }
  // No non-yaochuu tiles
  for (const key of Object.keys(counts)) {
    if (!yaochuuKeys.includes(key)) return false;
  }
  return doubleCount === 1;
}

// 面子分解: 将14张牌分解为 4面子 + 1雀头 的所有可能
export function findMentsuDecompositions(tiles) {
  const counts = {};
  tiles.forEach(t => {
    const key = `${t.suit}${t.rank}`;
    counts[key] = (counts[key] || 0) + 1;
  });

  const results = [];

  // Try each possible jantai (pair)
  for (const [jantaiKey, jantaiCount] of Object.entries(counts)) {
    if (jantaiCount < 2) continue;

    const remaining = { ...counts };
    remaining[jantaiKey] -= 2;
    if (remaining[jantaiKey] === 0) delete remaining[jantaiKey];

    const mentsus = extractMentsus(remaining);
    if (mentsus) {
      results.push({
        jantai: { suit: jantaiKey[0], rank: parseInt(jantaiKey.slice(1)) },
        mentsus,
      });
    }
  }

  return results;
}

// 从剩余牌中提取4个面子
function extractMentsus(counts) {
  const mentsus = [];
  const keys = Object.keys(counts).sort();

  // Process z tiles first (only koutsu possible)
  const zKeys = keys.filter(k => k[0] === 'z');
  for (const key of zKeys) {
    const count = counts[key];
    if (count === 3) {
      mentsus.push({ type: 'koutsu', suit: 'z', rank: parseInt(key.slice(1)) });
      delete counts[key];
    } else if (count > 0) {
      return null; // Can't form mentsu from leftover z tiles
    }
  }

  // Process m, p, s tiles
  const suitOrder = ['m', 'p', 's'];
  for (const suit of suitOrder) {
    const suitKeys = Object.keys(counts).filter(k => k[0] === suit).sort();
    const ranks = suitKeys.map(k => parseInt(k.slice(1)));

    // Try to decompose using greedy approach with backtracking
    const result = decomposeSuit(counts, suit, ranks);
    if (result === null) return null;
    mentsus.push(...result);
  }

  // Check if all tiles consumed
  if (Object.keys(counts).some(k => counts[k] > 0)) return null;

  return mentsus.length === 4 ? mentsus : null;
}

function decomposeSuit(counts, suit, ranks) {
  if (ranks.length === 0) return [];

  // Try koutsu first, then shuntsu
  const mentsus = [];
  const tempCounts = { ...counts };

  for (let r = 1; r <= 9; r++) {
    const key = `${suit}${r}`;
    while (tempCounts[key] >= 3) {
      mentsus.push({ type: 'koutsu', suit, rank: r });
      tempCounts[key] -= 3;
      if (tempCounts[key] === 0) delete tempCounts[key];
    }
  }

  // Then shuntsu
  for (let r = 1; r <= 7; r++) {
    const k1 = `${suit}${r}`;
    const k2 = `${suit}${r + 1}`;
    const k3 = `${suit}${r + 2}`;
    while (tempCounts[k1] > 0 && tempCounts[k2] > 0 && tempCounts[k3] > 0) {
      mentsus.push({ type: 'shuntsu', suit, ranks: [r, r + 1, r + 2] });
      tempCounts[k1]--;
      tempCounts[k2]--;
      tempCounts[k3]--;
      if (tempCounts[k1] === 0) delete tempCounts[k1];
      if (tempCounts[k2] === 0) delete tempCounts[k2];
      if (tempCounts[k3] === 0) delete tempCounts[k3];
    }
  }

  // Check remaining
  const remaining = Object.entries(tempCounts).filter(([k, v]) => k[0] === suit && v > 0);
  if (remaining.length > 0) {
    // Backtrack: try shuntsu-first approach
    return decomposeSuitShuntsuFirst({ ...counts }, suit);
  }

  return mentsus;
}

function decomposeSuitShuntsuFirst(counts, suit) {
  const mentsus = [];
  const tempCounts = { ...counts };

  // Shuntsu first
  for (let r = 1; r <= 7; r++) {
    const k1 = `${suit}${r}`;
    const k2 = `${suit}${r + 1}`;
    const k3 = `${suit}${r + 2}`;
    while (tempCounts[k1] > 0 && tempCounts[k2] > 0 && tempCounts[k3] > 0) {
      mentsus.push({ type: 'shuntsu', suit, ranks: [r, r + 1, r + 2] });
      tempCounts[k1]--;
      tempCounts[k2]--;
      tempCounts[k3]--;
      if (tempCounts[k1] === 0) delete tempCounts[k1];
      if (tempCounts[k2] === 0) delete tempCounts[k2];
      if (tempCounts[k3] === 0) delete tempCounts[k3];
    }
  }

  // Then koutsu
  for (let r = 1; r <= 9; r++) {
    const key = `${suit}${r}`;
    while (tempCounts[key] >= 3) {
      mentsus.push({ type: 'koutsu', suit, rank: r });
      tempCounts[key] -= 3;
      if (tempCounts[key] === 0) delete tempCounts[key];
    }
  }

  const remaining = Object.entries(tempCounts).filter(([k, v]) => k[0] === suit && v > 0);
  if (remaining.length > 0) return null;

  return mentsus;
}

export function checkYaku(handTiles, calls, roundInfo, doraTiles) {
  const yakuList = [];
  const allTiles = [...handTiles, ...calls.flatMap(c => c.tiles)];
  const isMenzenchin = calls.every(c => c.type === 'ankou' || c.type === 'kakan');

  // 断幺九: 没有任何幺九牌
  if (!allTiles.some(t => isYaochuuTile(t))) {
    yakuList.push({ name: '断幺九', han: 1 });
  }

  // 役牌: 自风、场风、三元牌的三刻
  // (will be checked in decomposition)

  // 立直/双立直 will be checked externally based on game state

  // 七对子
  if (isChitoi(handTiles) && isMenzenchin) {
    yakuList.push({ name: '七对子', han: 2 });
  }

  // 混一色: 只有1种数牌+字牌
  const suits = new Set(allTiles.filter(t => t.suit !== 'z').map(t => t.suit));
  if (suits.size === 1 && allTiles.some(t => t.suit === 'z')) {
    yakuList.push({ name: '混一色', han: isMenzenchin ? 3 : 2 });
  }

  // 清一色: 只有1种数牌，无字牌
  if (suits.size === 1 && !allTiles.some(t => t.suit === 'z')) {
    yakuList.push({ name: '清一色', han: isMenzenchin ? 6 : 5 });
  }

  // 对对和: 全刻子
  const decompositions = findMentsuDecompositions(handTiles);
  if (decompositions.length > 0) {
    for (const decomp of decompositions) {
      if (decomp.mentsus.every(m => m.type === 'koutsu')) {
        if (!yakuList.find(y => y.name === '对对和')) {
          yakuList.push({ name: '对对和', han: 2 });
        }
      }
    }
  }

  // 三暗刻
  const ankous = calls.filter(c => c.type === 'ankou');
  // Also check hand decomposition for hidden koutsu
  if (decompositions.length > 0) {
    for (const decomp of decompositions) {
      const handKoutsu = decomp.mentsus.filter(m => m.type === 'koutsu');
      const totalAnkou = ankous.length + handKoutsu.length;
      if (totalAnkou >= 3 && !yakuList.find(y => y.name === '三暗刻')) {
        yakuList.push({ name: '三暗刻', han: 2 });
      }
    }
  }

  // Dora bonus (not a yaku, just extra han)
  let doraHan = 0;
  for (const tile of allTiles) {
    for (const dora of doraTiles) {
      if (tile.suit === dora.suit && tile.rank === dora.rank) {
        doraHan++;

      if (tile.aka) doraHan++;
    }
  }
  if (doraHan > 0) {
    yakuList.push({ name: '宝牌', han: doraHan });
  }

  return yakuList;
}

export function countFu(decomposition, isTsumo, calls) {
  let fu = 20; // base fu

  // Jantai fu
  const jantai = decomposition.jantai;
  if (jantai.suit === 'z' || jantai.rank === 1 || jantai.rank === 9) {
    fu += 2;

      const isYaochuu = mentsu.suit === 'z' || mentsu.rank === 1 || mentsu.rank === 9;
      fu += isYaochuu ? 8 : 4;
    }
  }

  // Call fu
  for (const call of calls) {
    if (call.type === 'minpon') {
      const isYaochuu = call.tiles[0].suit === 'z' || call.tiles[0].rank === 1 || call.tiles[0].rank === 9;
      fu += isYaochuu ? 4 : 2;
    } else if (call.type === 'ankou') {
      const isYaochuu = call.tiles[0].suit === 'z' || call.tiles[0].rank === 1 || call.tiles[0].rank === 9;
      fu += isYaochuu ? 16 : 8;
    } else if (call.type === 'minchi') {
      fu += 2;
    } else if (call.type === 'minkan') {
      fu += call.tiles[0].suit === 'z' || [1, 9].includes(call.tiles[0].rank) ? 8 : 4;
    }
  }

  // Tsumo bonus
  if (isTsumo) fu += 2;

  // Round up to 10
  return Math.ceil(fu / 10) * 10;
}

// 计算点数
export function calculatePoints(han, fu, isDealer) {
  const basicPoints = fu * Math.pow(2, han + 2);

  // Minimum guarantees
  if (han === 1 && fu < 30) return { basic: 1000, dealer: 1500, child: 500 };
  if (han === 2 && fu < 20) return { basic: 1500, dealer: 2300, child: 700 };

  // Mangan at 5+ han or 4 han 40+ fu or 3 han 70+ fu
  if (han >= 5 || (han === 4 && fu >= 40) || (han === 3 && fu >= 70)) {
    return isDealer
      ? { basic: 12000, dealer: 12000 * 3 / 4, child: 12000 / 4 }
      : { basic: 8000, dealer: 8000, child: 8000 / 4 * 3 / 3 };
  }

  if (isDealer) {
    const total = Math.ceil(basicPoints * 6 / 100) * 100;
    const each = Math.ceil(basicPoints * 2 / 100) * 100;
    return { basic: basicPoints, dealer: each, child: each };
  } else {
    const dealerPay = Math.ceil(basicPoints * 6 / 100) * 100;
    const eachPay = Math.ceil(basicPoints * 2 / 100) * 100;
    return { basic: basicPoints, dealer: dealerPay, child: eachPay };
  }
}

// 向听数计算 (核心算法用于AI)
export function calculateShanten(tiles, calls) {
  // 4-面子1-雀头形式的最小向听数
  const mentsuNeeded = 4 - calls.length;
  const tilesInHand = tiles.length;

  // Simplified shanten calculation using mentsu count
  const counts = {};
  tiles.forEach(t => {
    const key = `${t.suit}${t.rank}`;
    counts[key] = (counts[key] || 0) + 1;
  });

  let mentsuCount = 0;
  let partialCount = 0;
  let jantaiCount = 0;

  // Count existing pairs (potential jantai)
  for (const [key, count] of Object.entries(counts)) {
    if (count >= 2) jantaiCount++;
  }

  // Count koutsu
  for (const [key, count] of Object.entries(counts)) {
    if (count >= 3) mentsuCount++;
  }

  // Count shuntsu (consecutive sequences)
  for (const suit of ['m', 'p', 's']) {
    for (let r = 1; r <= 7; r++) {
      const k1 = `${suit}${r}`;
      const k2 = `${suit}${r + 1}`;
      const k3 = `${suit}${r + 2}`;
      if (counts[k1] && counts[k2] && counts[k3]) {
        mentsuCount++;
        counts[k1]--;
        counts[k2]--;
        counts[k3]--;
      }
    }
  }

  // Count partial mentsu (2-tile combinations)
  // Pairs already counted as jantai candidates
  for (const suit of ['m', 'p', 's']) {
    for (let r = 1; r <= 8; r++) {
      const k1 = `${suit}${r}`;
      const k2 = `${suit}${r + 1}`;
      if (counts[k1] && counts[k2]) {
        partialCount++;
        counts[k1]--;
        counts[k2]--;
      }
    }
  }

  // Standard formula: 8 - 2*mentsu - partialCount - (jantai ? 1 : 0)
  // where mentsu includes calls
  const totalMentsu = mentsuCount + calls.length;
  const hasJantai = jantaiCount > 0;

  let shanten = 8 - 2 * totalMentsu - partialCount - (hasJantai ? 1 : 0);

  // Also check chitoi shanten
  const pairs = Object.values(counts).filter(v => v >= 2).length + jantaiCount;
  const chitoiShanten = 6 - pairs;

  // Kokushi shanten
  const yaochuuKeys = ['m1','m9','p1','p9','s1','s9','z1','z2','z3','z4','z5','z6','z7'];
  const yaochuuCount = yaochuuKeys.filter(k => counts[k]).length;
  const kokushiShanten = 13 - yaochuuCount - (yaochuuCount >= 1 ? 1 : 0);

  return Math.min(shanten, chitoiShanten, kokushiShanten);
}
```

```js
      { id: 3, name: '电脑3', points: INITIAL_POINTS, hand: null, discards: [], riichi: false, isDealer: false, calls: [], wind: WINDS[3] },
    ],
    wall: [],
    dora: [],
    doraIndicators: [],
    currentTurn: 0,
    lastDiscard: null,
    pendingActions: [],
    turnCount: 0,
  };
}

export function dealTiles(game) {
  const allTiles = createTileSet();
  const shuffled = xorshiftShuffle(allTiles, game.seed);

  const newGame = { ...game, phase: 'playing', wall: [] };

  // Each player gets 13 tiles, dealer gets 14 (4 rounds of 3 + 1 round of 1, then dealer draws 1)
  // Simplified: deal 13 to each, then dealer draws 1 more
  const hands = [[], [], [], []];
  let wallIndex = 0;

  // Deal in rounds of 4 (3 tiles each, 4 players) - traditional dealing
  for (let round = 0; round < 3; round++) {
    for (let player = 0; player < 4; player++) {
      for (let i = 0; i < 4; i++) {
        hands[player].push(shuffled[wallIndex++]);
      }
    }
  }
  // Final round: 1 tile each
  for (let player = 0; player < 4; player++) {
    hands[player].push(shuffled[wallIndex++]);
  }

  // Set hands
  newGame.players = game.players.map((p, i) => ({
    ...p,
    hand: createHand(sortHand(hands[i])),
    discards: [],
    riichi: false,
    calls: [],
  }));

  // Dora indicator (first of the dead wall)
  const doraIndicator = shuffled[wallIndex];
  newGame.doraIndicators = [doraIndicator];
  wallIndex++;

  // Remaining is the wall (living wall for drawing)
  newGame.wall = shuffled.slice(wallIndex);

  // Dealer draws first tile
  const dealerDraw = newGame.wall.shift();
  newGame.players[0].hand = { ...newGame.players[0].hand, drawnTile: dealerDraw };
  newGame.currentTurn = 0;
  newGame.turnCount = 0;

  return newGame;
}

export function nextTurn(game) {
  const nextPlayer = (game.currentTurn + 1) % 4;
  const drawTile = game.wall.shift();

  if (!drawTile) {
    // Wall exhausted - ryuukyoku (flowing draw)
    return { ...game, phase: 'result', resultType: 'ryuukyoku' };
  }

  const newPlayers = [...game.players];
  newPlayers[nextPlayer] = {
    ...newPlayers[nextPlayer],
    hand: { ...newPlayers[nextPlayer].hand, drawnTile: drawTile },
  };

  return {
    ...game,
    players: newPlayers,
    currentTurn: nextPlayer,
    turnCount: game.turnCount + 1,
    lastDiscard: null,
    pendingActions: [],
  };
}

export function discardTile(game, playerId, tileId) {
  const player = game.players[playerId];
  const hand = player.hand;

  // Find the tile being discarded
  let discardedTile;
  let newHand;

  if (hand.drawnTile && hand.drawnTile.id === tileId) {
    discardedTile = hand.drawnTile;
    newHand = { ...hand, drawnTile: null };
  } else {
    discardedTile = hand.tiles.find(t => t.id === tileId);
    newHand = {
      ...hand,
      tiles: hand.tiles.filter(t => t.id !== tileId),
      drawnTile: hand.drawnTile, // keep drawn tile if removing from main hand
    };
  }

  const newDiscards = [...player.discards, discardedTile];
  const newPlayers = [...game.players];
  newPlayers[playerId] = {
    ...player,
    hand: newHand,
    discards: newDiscards,
  };

  return {
    ...game,
    players: newPlayers,
    lastDiscard: discardedTile,
    pendingActions: [],
  };
}

export function advanceRound(game, dealerWon) {
  const ri = game.roundInfo;

  if (dealerWon) {
    // Dealer continues, honba +1
    return {
      ...game,
      roundInfo: {
        ...ri,
        honba: ri.honba + 1,
        riichiSticks: 0,
      },
    };
  }

  // Dealer rotates
  const nextNumber = ri.number + 1;
  let nextWind = ri.wind;

  // Check if we need to change wind
  if (nextNumber > 4) {
    if (ri.wind === '東') {
      nextWind = '南';
      nextNumber = 1;
    } else {
      // 南4局结束 - game over
      return { ...game, phase: 'end' };
    }
  }

  return {
    ...game,
    roundInfo: {
      wind: nextWind,
      number: nextNumber,
      honba: 0,
      riichiSticks: 0,
    },
  };
}
```

```js
// 手牌评估器 - 向听数计算与打牌评分

import { calculateShanten as coreShanten } from '../../core/validator.js';

export function calculateShanten(tiles, calls) {
  return coreShanten(tiles, calls);
}

// 找出所有能降低向听数的有效牌
export function findEffectiveTiles(tiles, calls, allKnownTiles) {
  const currentShanten = calculateShanten(tiles, calls);
  const effective = [];

  // Try adding each possible tile type and check if shanten decreases
  const tileTypes = new Set(tiles.map(t => `${t.suit}${t.rank}`));
  const allTypes = ['m','p','s'].flatMap(s =>
    [1,2,3,4,5,6,7,8,9].map(r => `${s}${r}`)
  ).concat(['z1','z2','z3','z4','z5','z6','z7']);

  for (const typeKey of allTypes) {
    // Check if this tile type still has remaining copies
    const usedCount = allKnownTiles.filter(t =>
      `${t.suit}${t.rank}` === typeKey
    ).length;
    const maxCount = 4;
    if (usedCount >= maxCount) continue;

    const suit = typeKey[0];
    const rank = parseInt(typeKey.slice(1));
    const testTile = { suit, rank, aka: false, id: -1 };
    const testHand = [...tiles, testTile];
    const newShanten = calculateShanten(testHand, calls);

    if (newShanten < currentShanten) {
      const remaining = maxCount - usedCount;
      effective.push({ suit, rank, remaining, shantenReduction: currentShanten - newShanten });
    }
  }

  return effective;
}

// 对每张可打的牌评分
export function scoreDiscard(handTiles, calls, allKnownTiles) {
  const scores = [];

  for (let i = 0; i < handTiles.length; i++) {
    const tile = handTiles[i];
    const remaining = [...handTiles];
    remaining.splice(i, 1);

    const shanten = calculateShanten(remaining, calls);
    const effectiveTiles = findEffectiveTiles(remaining, calls, allKnownTiles);

    // Score: lower shanten is better, more effective tiles is better
    const effectiveCount = effectiveTiles.reduce((sum, e) => sum + e.remaining, 0);
    const score = -shanten * 100 + effectiveCount * 10;

    // Dora proximity bonus
    // (simplified: tiles near dora ranks get small bonus)

    scores.push({ tile, score, shanten, effectiveCount });
  }

  // Sort by score descending (best discard = lowest score, so we want max score = best to KEEP)
  scores.sort((a, b) => b.score - a.score);

  return scores;
}

// 选择最优打牌
export function chooseBestDiscard(handTiles, calls, allKnownTiles) {
  const scores = scoreDiscard(handTiles, calls, allKnownTiles);
  // The tile with the lowest score is the best to discard
  return scores[scores.length - 1].tile;
}
```

```js
import { calculateShanten, chooseBestDiscard, findEffectiveTiles } from './evaluator.js';
import { isMenzenchin, canRiichi } from '../../core/rules.js';
import { getAllHandTiles } from '../../core/hand.js';
import { isWinningHand } from '../../core/validator.js';

export function aiDecideDiscard(hand, allKnownTiles, gameState) {
  const allTiles = getAllHandTiles(hand);
  const shanten = calculateShanten(allTiles, hand.calls);

  // If winning, don't discard (should have been caught earlier)
  if (shanten === -1) return null;

  // Check for defensive play (someone else is riichi)
  const opponentRiichi = gameState.players && gameState.players.some(
    p => p.id !== 0 && p.riichi
  );

  if (opponentRiichi) {
    return defensiveDiscard(hand, allKnownTiles, gameState);
  }

  // Normal offensive play
  return chooseBestDiscard(allTiles, hand.calls, allKnownTiles);
}

export function aiDecideRiichi(hand, playerInfo) {
  const allTiles = getAllHandTiles(hand);
  const shanten = calculateShanten(allTiles, hand.calls);

  if (!isMenzenchin(hand.calls)) return false;
  if (shanten > 0) return false;
  if (playerInfo.points < 1000) return false;

  // Tenpai - consider riichi strength
  // With good effective tiles (many remaining copies), riichi is strong
  const effective = findEffectiveTiles(allTiles, hand.calls, allTiles);
  const totalEffective = effective.reduce((sum, e) => sum + e.remaining, 0);

  // Strong hand: many effective tiles remaining -> riichi
  if (totalEffective >= 4) return true;

  // Weak tenpai: few effective tiles -> might still riichi if late turn
  return true; // In difficult mode, always riichi when tenpai and menzenchin
}

// 防守打牌: 当对手立直时优先打安全牌
function defensiveDiscard(hand, allKnownTiles, gameState) {
  const allTiles = getAllHandTiles(hand);

  // Safe tiles: tiles already discarded by the riichi player (现物)
  const riichiPlayer = gameState.players.find(p => p.id !== 0 && p.riichi);
  if (!riichiPlayer) return chooseBestDiscard(allTiles, hand.calls, allKnownTiles);

  const safeKeys = new Set(riichiPlayer.discards.map(t => `${t.suit}${t.rank}`));

  // Find safe tiles in our hand
  const safeTiles = allTiles.filter(t => safeKeys.has(`${t.suit}${t.rank}`));

  if (safeTiles.length > 0) {
    // Among safe tiles, discard the least useful one
    return chooseBestDiscard(safeTiles, hand.calls, allKnownTiles);
  }

  // No safe tiles - check suji (筋牌)
  const sujiTiles = findSujiTiles(allTiles, riichiPlayer.discards);
  if (sujiTiles.length > 0) {
    return chooseBestDiscard(sujiTiles, hand.calls, allKnownTiles);
  }

  // No safe or suji tiles - fall back to normal strategy but prefer lowest-value tiles
  return chooseBestDiscard(allTiles, hand.calls, allKnownTiles);
}

function findSujiTiles(handTiles, discards) {
  // 筋牌: if someone discarded m4, then m1 and m7 are suji
  const sujiKeys = new Set();
  for (const d of discards) {
    if (d.suit === 'z') continue;
    // Shuntsu middle tile determines suji
    sujiKeys.add(`${d.suit}${d.rank - 3}`);
    sujiKeys.add(`${d.suit}${d.rank + 3}`);
  }
  return handTiles.filter(t => sujiKeys.has(`${t.suit}${t.rank}`));
}
```

```js
// AI 吃碰杠/和牌的反应决策

import { calculateShanten } from './evaluator.js';
import { getAllHandTiles } from '../../core/hand.js';
import { canChi, canPon, canMinkan, canAnkan, canKakan } from '../../core/hand.js';
import { isWinningHand } from '../../core/validator.js';

export function aiDecideReaction(hand, discardedTile, playerId, gameState) {
  const allTiles = getAllHandTiles(hand);
  const shanten = calculateShanten(allTiles, hand.calls);

  const actions = [];

  // 1. Check if can win (highest priority)
  const testHand = [...allTiles, discardedTile];
  if (isWinningHand(testHand) && !isFuriten(hand, discardedTile, gameState)) {
    actions.push({ type: 'ron', priority: 5 });
  }

  // 2. Check pon
  if (canPon(hand, discardedTile)) {
    const testHandAfterPon = allTiles.filter(t =>
      !(t.suit === discardedTile.suit && t.rank === discardedTile.rank)
    ).slice(0, allTiles.length - 2);
    const newShanten = calculateShanten(testHandAfterPon, [...hand.calls, { type: 'minpon' }]);
    if (shanten <= 2 || newShanten < shanten) {
      actions.push({ type: 'pon', priority: 3 });
    }
  }

  // 3. Check minkan
  if (canMinkan(hand, discardedTile)) {
    if (shanten <= 1) {
      actions.push({ type: 'minkan', priority: 4 });
    }
  }

  // 4. Check chi (only for next player)
  const nextPlayer = (gameState.currentTurn + 1) % 4;
  if (playerId === nextPlayer) {
    const chiOptions = canChi(hand, discardedTile, null);
    if (chiOptions.length > 0 && shanten <= 2) {
      actions.push({ type: 'chi', priority: 2, options: chiOptions });
    }
  }

  // Sort by priority
  actions.sort((a, b) => b.priority - a.priority);

  return actions.length > 0 ? actions[0] : { type: 'skip', priority: 0 };
}

export function aiDecideTsumoAction(hand, gameState) {
  const allTiles = getAllHandTiles(hand);

  // Check tsumo win
  if (isWinningHand(allTiles) && hand.drawnTile) {
    return { type: 'tsumo' };
  }

  // Check ankan
  const ankans = canAnkan(hand);
  if (ankans.length > 0) {
    const shanten = calculateShanten(allTiles, hand.calls);
    // Only ankan if it doesn't worsen shanten significantly
    const newCalls = [...hand.calls, { type: 'ankou' }];
    const newShanten = calculateShanten(
      allTiles.filter(t => !(t.suit === ankans[0].suit && t.rank === ankans[0].rank)),
      newCalls
    );
    if (newShanten <= shanten) {
      return { type: 'ankan', suit: ankans[0].suit, rank: ankans[0].rank };
    }
  }

  // Check kakan
  const kakans = canKakan(hand);
  if (kakans.length > 0) {
    return { type: 'kakan', call: kakans[0] };
  }

  // Otherwise discard
  return { type: 'discard' };
}

// 振听检查 (简化版)
function isFuriten(hand, tile, gameState) {
  // Check if player has previously discarded the same tile type
  // This means they can't win with that tile via ron
  const player = gameState.players.find(p => p.hand === hand);
  if (!player) return false;

  return player.discards.some(d =>
    d.suit === tile.suit && d.rank === tile.rank
  );
}
```

```jsx
    const allKnown = game.players.flatMap(p => p.discards);
    const discard = aiDecideDiscard(currentPlayer.hand, [...allTiles, ...allKnown], game);

    if (!discard) return;

    // Check riichi
    const shanten = calculateShanten(allTiles, currentPlayer.hand.calls);
    const shouldRiichi = !currentPlayer.riichi && shanten === 0 && isMenzenchin(currentPlayer.hand.calls);

    let newGame = discardTile(game, game.currentTurn, discard.id);

    if (shouldRiichi) {
      newGame.players[game.currentTurn] = {
        ...newGame.players[game.currentTurn],
        riichi: true,
        points: newGame.players[game.currentTurn].points - 1000,
      };
      newGame.roundInfo = {
        ...newGame.roundInfo,
        riichiSticks: newGame.roundInfo.riichiSticks + 1,
      };
    }

    // Check if other players can react
    checkReactions(newGame);
  }, []);

  const checkReactions = useCallback((game) => {
    const discarded = game.lastDiscard;
    if (!discarded) {
      // No discard, move to next turn
      const next = nextTurn(game);
      setGameState(next);
      if (next.players[next.currentTurn].id !== 0) {
        scheduleAiAction(next);
      }
      return;
    }

    // Check each player for possible reactions
    const actions = [];

    for (let i = 0; i < 4; i++) {
      if (i === game.currentTurn) continue; // Skip the discarding player

      const player = game.players[i];
      const reaction = aiDecideReaction(player.hand, discarded, i, game);

      if (reaction.type === 'ron') {
        handleWin(game, i, false);
        return;
      }

      if (reaction.type !== 'skip') {
        if (i === 0) {
          // Player needs to decide
          actions.push(reaction);
        } else {
          // AI auto-decides
          executeReaction(game, i, reaction);
          return;
        }
      }
    }

    // No reactions - next turn
    const next = nextTurn(game);
    setGameState(next);
    if (next.players[next.currentTurn].id !== 0) {
      scheduleAiAction(next);
    }
  }, []);

  const playerDiscard = useCallback((tile) => {
    if (!gameState || gameState.currentTurn !== 0) return;

    const allTiles = getAllHandTiles(gameState.players[0].hand);
    if (isWinningHand(allTiles)) {
      // Player can tsumo
      setGameState({ ...gameState, pendingActions: [{ type: 'tsumo' }] });
      return;
    }

    const newGame = discardTile(gameState, 0, tile.id);
    setGameState(newGame);
    checkReactions(newGame);
  }, [gameState]);

  const playerAction = useCallback((action) => {
    if (action.type === 'skip') {
      const next = nextTurn(gameState);
      setGameState(next);
      if (next.players[next.currentTurn].id !== 0) {
        scheduleAiAction(next);
      }
      return;
    }

    if (action.type === 'tsumo') {
      handleWin(gameState, 0, true);
      return;
    }

    if (action.type === 'ron') {
      handleWin(gameState, 0, false);
      return;
    }

    // Other actions (chi, pon, kan) - simplified handling
    const next = nextTurn(gameState);
    setGameState(next);
  }, [gameState]);

  const handleWin = useCallback((game, winnerId, isTsumo) => {
    const winner = game.players[winnerId];
    const allTiles = getAllHandTiles(winner.hand);
    const doraTiles = game.doraIndicators.map(ind => doraFromIndicator(ind));

    const yaku = checkYaku(allTiles, winner.hand.calls, game.roundInfo, doraTiles);
    const totalHan = countHan(yaku);

    let fu = 30; // simplified default
    if (totalHan <= 4) {
      const decompositions = findMentsuDecompositions(allTiles);
      if (decompositions.length > 0) {
        fu = countFu(decompositions[0], isTsumo, winner.hand.calls);
      }
    }

    const points = calculatePoints(totalHan, fu, winner.isDealer);

    const pointsChange = {};
    game.players.forEach(p => {
      if (p.id === winnerId) {
        pointsChange[p.name] = isTsumo
          ? (p.isDealer ? points.dealer * 3 : points.child * 3)
          : (p.isDealer ? points.dealer * 3 : points.dealer + points.child * 2);
      } else {
        pointsChange[p.name] = isTsumo
          ? -(p.isDealer ? points.dealer : points.child)
          : -(p.id === game.lastDiscardPlayerId ? points.dealer : points.child);
      }
    });

    setResult({
      type: isTsumo ? 'tsumo' : 'ron',
      winner: winner.name,
      loser: isTsumo ? null : game.players[game.lastDiscardPlayerId]?.name,
      yaku,
      totalHan,
      fu,
      pointsChange,
    });
  }, []);

  const continueGame = useCallback(() => {
    setResult(null);
    const dealerWon = gameState.players[0].isDealer && gameState.players[0].id === gameState.lastWinnerId;
    const nextRound = advanceRound(gameState, dealerWon);

    if (nextRound.phase === 'end') {
      setGameState(nextRound);
      return;
    }

    const newSeed = gameState.seed + 1;
    const newGame = createGame(newSeed);
    const dealt = dealTiles(newGame);
    setGameState(dealt);

    if (dealt.players[dealt.currentTurn].id !== 0) {
      scheduleAiAction(dealt);
    }
  }, [gameState]);

  return { gameState, result, startGame, playerDiscard, playerAction, continueGame };
}

import { findMentsuDecompositions } from '../core/validator';
```
